Overview

初始化组件

                    KISSY.use("gallery/auth/1.4/index", function (S, Auth) {
                        var form = new Auth('#J_form_eg1', {
                            msg:{
                                tpl:'
{msg}
',
                                args:{
                                    prefixCls:'under'
                                }
                            }
                        });
                    })

自定义验证规则

以自定义银行卡号验证规则为例

                //在全局上下文中添加一个自定义规则
                form.register('card', function (value) {
                    return value.length > 3;
                });
                form.add('#card', {
                    rules:{
                        card: {
                            success:'输入卡号正确',
                            error:'卡号必须大于3位'
                        }
                    },
                    style:{
                        success:'attention',
                        error:'error'
                    }
                });

手动验证某个字段

                    form.getField('card').validate();

手动验证表单所有规则

                KISSY.Event.on('#J_form_eg1', "submit", function () {
                    return form.validate();
                });

                form.on('validate', function(e) {
                    alert(e.msg);
                });
